final decoration = BoxDecoration(
	color: Colors.red,

class MyList extends StatefullWidget {
	State createState() => MyListState();
}

class MyListState extends State {
  int index = 0;

  build(_) => CrawlStickerSurface(
    child: ListView(
	children: List.generate((i) => InkWell(
		onTap: () => setState(() => index = i),
		child: Row(
			children: [StickerWidget(decoration: decoration, show: i == index), Text('Item $i')]
),
)),
),	
  );
}
